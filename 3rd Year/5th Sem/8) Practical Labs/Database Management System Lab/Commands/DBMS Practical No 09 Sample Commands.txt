-- Create Database
CREATE DATABASE joins_demo_large;
USE joins_demo_large;

-- Create Departments Table
CREATE TABLE departments (
   dept_id INT PRIMARY KEY,
   dept_name VARCHAR(50),
   location VARCHAR(50)
);

-- Insert Departments

(5, 'HR', 'Block E');

-- Show Departments Table
SELECT * FROM departments;

-- Create Employees Table
CREATE TABLE employees (
   emp_id INT PRIMARY KEY,
   name VARCHAR(50),
   dept_id INT,
   salary DECIMAL(10,2),
   FOREIGN KEY (dept_id) REFERENCES departments(dept_id)
);

-- Insert Employees

-- Show Employees Table

-- Create Projects Table
CREATE TABLE projects (
   project_id INT PRIMARY KEY,
   emp_id INT,
   project_name VARCHAR(50),
   FOREIGN KEY (emp_id) REFERENCES employees(emp_id)
);

-- 1. INNER JOIN (Employees with Departments)
SELECT e.emp_id, e.name, d.dept_name, e.salary
FROM employees e
INNER JOIN departments d ON e.dept_id = d.dept_id;

-- 2. LEFT JOIN (All Employees, with Dept if available)
SELECT e.emp_id, e.name, d.dept_name, e.salary
FROM employees e
LEFT JOIN departments d ON e.dept_id = d.dept_id;

-- 3. RIGHT JOIN (All Departments, with Employees if available)
SELECT e.emp_id, e.name, d.dept_name
FROM employees e
RIGHT JOIN departments d ON e.dept_id = d.dept_id;

-- 4. FULL OUTER JOIN (Simulated with UNION in MySQL)
SELECT e.emp_id, e.name, d.dept_name
FROM employees e
LEFT JOIN departments d ON e.dept_id = d.dept_id
UNION
SELECT e.emp_id, e.name, d.dept_name
FROM employees e
RIGHT JOIN departments d ON e.dept_id = d.dept_id;

-- 5. MULTI-TABLE JOIN (Employees with Projects and Departments)
SELECT e.name, p.project_name, d.dept_name
FROM employees e
INNER JOIN projects p ON e.emp_id = p.emp_id
INNER JOIN departments d ON e.dept_id = d.dept_id;

-- 6. CROSS JOIN (All combinations of Employees × Departments)
SELECT e.name, d.dept_name
FROM employees e
CROSS JOIN departments d;
