print("linear Search Algorithm")
def linear_search(list1,key,n):
    for i in range(0,n):
        if list1[i]==key:
            return i
    return -1
list1=[1,2,4,3,7,5,6]
key=2
n=len(list1)
res=linear_search(list1,key,n)
if res==-1:
    print("Element not Found")

else:
    print("Element found at index:",res)
