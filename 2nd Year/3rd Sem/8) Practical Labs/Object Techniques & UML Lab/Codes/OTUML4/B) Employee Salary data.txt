#include <iostream>
#include <string>
using namespace std;

class Employee {
private:
    string name;
    int age;
    string salary;

public:
    Employee(string n, int a, string s) {
        name = n;
        age = a;
        salary = s;
    }

    void display() {
        cout << "Name: " << name << ", Age: " << age << ", salary: " << salary << endl;
    }
};

int main() {
    
   Employee e1("John", 20, "100000");
   Employee e2("Jiya", 25, "20000");
   Employee e3("Alice",30, "30000");
  
    e1.display();
    e2.display();
    e3.display();
   
    return 0;
}
